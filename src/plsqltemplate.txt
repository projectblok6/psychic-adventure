CREATE OR REPLACE TRIGGER %triggername%
%triggerevent% on %tablename% FOR EACH ROW
begin
  declare
    l_oper varchar2 ( 3 );
    l_error_stack varchar2 ( 4000 )
  if inserting then
    l_oper := 'INS';
  elsif updating then
    l_oper := 'UPD';
  elsif deleting then
    l_oper := 'DEL';
  end if;
  %startloop%
  declare
    l_passed boolean := true;
    %declarations%
  begin
    if l_oper in (%triggerevents%) then
      %selectstatements%
      l_passed := %operator%
      if not l_passed then
        l_error_stack := l_error_stack || '%errormessage%';
      end if;
    end if;
  end;
  %endloop%
  if l_error_stack is not null then
    raise_application_error ( -20800, l_error_stack );
  end if;
end %triggername%;
